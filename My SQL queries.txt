#1 Year wise loan amount Stats
SELECT
  YEAR(STR_TO_DATE(CONCAT('01-', issue_d), '%d-%b-%y')) AS loan_year,
  COUNT(*) AS total_loans,
  SUM(loan_amnt) AS total_loan_amount,
  ROUND(AVG(loan_amnt), 2) AS avg_loan_amount
FROM finance_1
WHERE issue_d IS NOT NULL
GROUP BY loan_year
ORDER BY loan_year;
#2 Grade and sub grade wise revol_bal
SELECT
    f1.grade,
    f1.sub_grade,
    SUM(f2.revol_bal) AS total_revol_balance
FROM finance_1 f1
JOIN finance_2 f2
    ON f1.id = f2.id
GROUP BY f1.grade, f1.sub_grade
ORDER BY f1.grade, f1.sub_grade;
DESCRIBE finance_1;
DESCRIBE finance_2;
#3 Total Payment for Verified Status Vs Total Payment for Non Verified Status
SELECT
  verification_status,
  SUM(funded_amnt) AS total_funded_amount
FROM finance_1
GROUP BY verification_status;
#4 State wise and month wise loan status
SELECT
  addr_state,
  loan_status,
  COUNT(*) AS total_loans
FROM finance_1
GROUP BY addr_state, loan_status
ORDER BY addr_state, total_loans DESC;
#5 Home ownership Vs last payment date stats
SELECT
  home_ownership,
  SUM(funded_amnt) AS total_funded_amount,
  COUNT(*) AS total_loans
FROM finance_1
GROUP BY home_ownership;
